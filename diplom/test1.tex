\section{Постановка задачи} 

%%%%%%%%%%%%%%%%%%%%%%%% Постановка задачи  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{

\frametitle{Модель гибридной системы} 
\begin{equation}
\left\{
\begin{aligned}
&\left[
 \begin{aligned}
    &\left\{
		    \begin{aligned}
		        &\dot x^{(1)} = A^{(1)}(t)x^{(1)}+B^{(1)}(t)u^{(1)}(t)+C^{(1)}(t)v(t);\\
		        &x(t_0)\in\mathcal{E}(x_0,X_0)=\mathcal{X}^0;\\
		        &u^{(1)}(t)\in\mathcal{E}(p^{(1)}(t),P^{(1)}(t))\subset\mathcal{P}[t_0,t_1];\\
		        &v^{(1)}(t)\in\mathcal{E}(w^{(1)}(t),W^{(1)}(t))\subset\mathcal{V}[t_0,t_1];\\
		    \end{aligned}
    \right.\\
    \\
    &\left\{
		    \begin{aligned}
		        &\dot x^{(2)} = A^{(2)}(t)x^{(2)}+B^{(2)}(t)u^{(2)}(t)+C^{(2)}(t)f(t)^{(2)};\\
		        &x(\tau)\in\mathcal{X}^{(1)}(\tau,t_0)\cap H;\\
		        &u^{(2)}(t)\in\mathcal{E}(p^{(2)}(t),P^{(2)}(t))\subset\mathcal{P}[t_0,t_1];\\
		        &v^{(2)}(t)\in\mathcal{E}(w^{(2)}(t),W^{(2)}(t))\subset\mathcal{V}[t_0,t_1];\\
		    \end{aligned}
    \right.
	\end{aligned}
\right.
\\
&H=\{x\in\mathbb R^n\colon \ts{x,c}=\gamma\}\mbox{--- гиперплоскость переключения};\\
%&A(t)\in \mathbb{R}^{n\times n},\,B(t)\in \mathbb{R}^{n\times m},\,C(t)\in \mathbb{R}^{n\times k};\;A(t),B(t),C(t) \in  L_1[t_0,t_1];\\
&\tau:\;\ts{x^{(1)}(\tau),c}-\gamma=0,\;\tau\in[t_0,t_1]
\end{aligned}
\right.
\label{eq:mainsystem}
\end{equation}


}

%%%%%%%%%%%%%%%%%%%%%%%% трансверсальность %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame} \frametitle{Условие на переключение}

При переключении гибридная система должна удовлетворять условию односторонней проницаемости (трансверсальности), что равносильно выполнению системы соотношений: 
% то есть чтобы при переходе через плоскость $H$ в момент $\tau$ система
\begin{equation}
\left\{
\begin{aligned}
	&\left[
		\begin{aligned}
			&\ts{\dot x,c}\neq 0;\\
			&\ts{x,c} - \gamma \neq 0\\
		\end{aligned}
	\right.\\
	&\mbox{sign}(\ts{\dot x^{(1)}(\tau),c})=\mbox{sign}(\ts{\dot x^{(2)}(\tau),c}).
\end{aligned}
\right.
\label{eq:perehod}
\end{equation}

Это означает, что любая траектория гибридной системы не может касаться гиперплоскости, но пересекает эту плоскость только один раз.

\end{frame}

\frame{  \frametitle{Множество достижимости}

\begin{define}
Множеством достижимости $\mathcal {X}[t,t_0]$  для задачи \eqref{eq:mainsystem} называется
$$
\mathcal {X}[t]=\big\{x^* \mid\exists u_1(\cdot)\in\mathcal{P}^{(1)},\exists u_2(\cdot)\in\mathcal{P}^{(2)}:
$$
$$
\forall v_1(\cdot)\in\mathcal{V}^{(1)},\forall v_2(\cdot)\in\mathcal{V}^{(2)}:
$$
$$
x(t_0,t_1,x^*)\in\mathcal{X}_0, x(t,t_0,x^*)\cap H \neq \emptyset \big\}
$$

\end{define}
}

\subsection{Применение методов многозначного анализа}

\frame{   \frametitle{Множество достижимости до переключения}

$$
V_1[t_0,t]=\left\{ \intd{t_0}{t}v_1(s)ds \mid \forall v_1(\cdot)\in\mathcal{V}^1[t_0,t_1]\right\}  
$$

$$
\mathcal{C}[t]=\mathcal{X}_0\dot-V_1[t,t_0]\;\; \mbox{-- множество гарантированного попадания}
$$
$$
p^\alpha(\cdot)\in \mathcal{P}^{(1)}: \;\;\;\; \mathcal{C}^\alpha_H[\tau]= \left\{ (\mathcal{C}[\tau]+p^\alpha(t_0,\tau)) \cap H \right\}
$$

$$
T^\alpha=\left\{\tau\mid \mathcal{C}^\alpha_H[\tau]\neq\emptyset \right\}\overset{\underset{\mbox{(трансверсальность)}}{}}{=}[\tau^\alpha_1,\tau^\alpha_2]
$$


}





%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% после переключения  %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{ \frametitle{Множество достижимости при переключении }

$$
\mathcal{P}_2[t_0,t_1]=\left\{\intd{t_0}{t_1}p_2(s)ds\mid \forall p_2(\cdot)\in \mathcal{P}^{(2)}\right\}
$$

$$
\mathcal{X}^\alpha_2[t_1]=\bigcap_{v_2\in \mathcal{V}_2}\bigcup_{\tau\in T^\alpha}(\mathcal{C}^\alpha_H[\tau]+v_2(\tau,t_1)+\mathcal{P}_2[\tau,t_1])
$$

$$
\mathcal{X}^{(2)}[t_1]=\bigcup_{p^\alpha\in \mathcal{P}_1}\mathcal{X}^\alpha_2[t_1]
$$

}

\frame{ \frametitle{Множество достижимости после переключения }
$$
x(t_1)=G_2(\tau,t_1)x(\tau)+\intd{\tau}{t_1}G_2(t_1,s)B_2(s)u_2(s)ds+\intd(\tau){t_1}G_2(t,s)C_2(s)v(s)ds
$$

$$
V_2[t_0,t]=\left\{ \intd{t_0}{t}v_2(s)ds \mid \forall v_2(\cdot)\in\mathcal{V}^2[t_0,t_1]\right\}  
$$

$$
\xi(\tau)\overset{def}{=}G_2(\tau,t_1)x(\tau)+\intd{t_1}{\tau}G_2(t_1,s)B_2(s)u_2(s)ds
$$

{\center Появление фазовых ограничений.}

$$
Z(\tau)=\{\xi(\tau)\mid  \left(\xi(\tau)+V_2[t,\tau]\right)\cap H \subset \mathcal{C}^\alpha_H[\tau] \},\; \tau\in T^\alpha
$$

}

\frame{ \frametitle{Решение задачи с фазовыми ограничениями}

Задача на поиск множества достижимости при фазовых ограничениях, но уже без помехи.
$$
\mathcal{X}^\alpha_2[t_1]=\{x(t_1) \mid  \exists u_2: G(\tau,t_1)x(t_1)+\intd{t_1}{\tau}B_2(s)u_2(s)ds \in Z(\tau), \forall \tau\in T^\alpha\}
$$

Функция цены при вычислении ограничений
$$
V(t,x)=\min_{u_2}\left(d^2( x(\tau,t,x), \mathcal{C}^\alpha_H)+\intd{\tau^\alpha_1}{\tau^\alpha_2}d^2 (x(s,u_2(\cdot)),Z(s)) ds\right)
$$
$$
V(t,x)=\max\big\{\intd{\tau}{t}\ts{w(t,\tau,\ell),x}ds - \intd{\tau}{t}\supfn{w(s,\tau,ell)}{B_2(s)\mathcal{P}^2(s)}ds-
$$
$$
-\intd{\tau}{t}\supfn{\lambda(s)}{Z(s)}ds\mid \;\ts{\ell,\ell}=1,\;\lambda(\cdot)\in V[\tau,t]\big\}
$$
}




\subsection{Применение методов выпуклого анализа}

\frame{ \frametitle{Функция цены, исходный вид}

$$
V(t,x)=\min_{u_1,u_2}\max_{v_1,v_2}\min_{\tau}\left\{ d^2(x_0{\mid_{x(t)=x}},\mathcal{X}_0) + \left(\ts{x(\tau),c}-\gamma\right)^2 \right\}
$$

$$
u_1(\cdot)\in\mathcal{P}^{(1)}[t_0,t_1],\;u_2(\cdot)\in\mathcal{P}^{(1)}[t_0,t_1],
$$

$$
v_1(\cdot)\in\mathcal{V}^{(1)}[t_0,t_1],\;v_2(\cdot)\in\mathcal{V}^{(2)}[t_0,t_1],
$$

$$
\tau\in [t_0,t_1]
$$

%$$

%$$

}

%слайд
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{ \frametitle{ Нелинейность по времени и теорема о минимаксе}

Для перехода к опорным функциям требуется менять местами порядок минимумов и максимумов, но
функционал вида 
$$
T(\tau,\Psi(\cdot))=\intd{t}{\tau}\Psi(s)ds 
$$
является нелинейным по $\tau$. Используя класс функций ограниченной вариации $\phi(w)\in V[t_0,t_1] $ 
преобразуем к виду 
$$
T(\phi(w),\Psi(\cdot))=\intd{t_0}{t}  d\phi(w) \int\limits_t^{w}\Psi(w)ds.
$$
Теперь  функционал $T(\phi(\cdot),\Psi(\cdot))$ является линейным по $\phi(\cdot)$ и можно воспользоваться теоремой о минимаксе.

}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\frame{ \frametitle{Функция цены, преобразования}

$$
S(t_0,t_1)=\intd{t_0}{t_1}\left\{  G^T_2(w,t)G^T_1(t_0,w)\ell(w) + \mu(w) G_2^T(w,t)c \right\}\,d\phi(w), 
$$

$$
S_1(t_0,t_1)=\intd{t_0}{t_1}\left\{ G^T_1(t_0,w)  \ell(w) \right\}\,d\phi(w),
$$

$$
K(\ell,\mu,\mathcal{X}_0)=\intd{t_0}{t} \left[\frac{\mu(w)^2}{4} + \rho(\ell(w)\mid\mathcal{X}_0) + \frac{\|\ell(w)\|^2}{4} \right]\,d\phi(w)
$$

}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\frame{ \frametitle{Функция цены, конечное выражение}

$$
 V(t,x)=\min_{\phi(\cdot)} \max_{\ell(\cdot)}\max_{\mu(\cdot)} \bigg\{  S^T(t_0,t)x - \intd{t_0}{t} \rho(S_1(s,t)\mid B_1(s)\mathcal{P}_1(s))\,ds - 
$$
$$
   - \intd{t_0}{t} \rho(S(t_0,s)\mid B_2(s)\mathcal{P}_2(s))\,ds - \mbox{conv}\big\{ K(\ell,\mu,\mathcal{X}_0) -
$$ 
$$
 -\intd{t_0}{t}\rho(-S_1(s,t)\mid C_1(s)\mathcal{V}_1(s))\,ds-\intd{t_0}{t}\rho(-S(t_0,s)\mid C_2(s)\mathcal{V}_2(s)) \,ds   \big\}\bigg\}
$$

}

\section{В заключение }


\frame{ \frametitle{Выводы}

%взято из введениия
В данной работе рассмотрены два подхода к решению задачи достижимости: 
\begin{itemize}
\item При помощи функций цены и методов выпуклого анализа
\item C помощью методов многозначного анализа.
\end{itemize} 

В первом случае множества достижимости представлены в виде множеств уровней (множеств Лебега) для специальных функций, а во втором методе указанные множества построены в явном виде, за счет применения операций с выпуклыми компактами. 
%Рассмотрены два примера в одномерном фазовом пространстве.

Решение данной задачи можно рассматривать как первый шаг к построению теории достижимости при неопределенности для гибридных систем с кусочно-линейной структурой в классах программных или позиционных управлений.
