\subsection{Пример 1а, $\rdim{1}$}


\begin{equation}
\label{eq:prim1a}
\left\{
\begin{aligned}
	&\left[
		\begin{aligned}
				&\dot x=u+v;  \\
				&\dot x=u;
		\end{aligned}
	 \right.\\
	&u(t)\in[p_1,p_2]=\mathcal{P};\\
	&v(t)\in[q_1,q_2]=\mathcal{Q};\\
	&\mathcal{X}_0=[a_1,a_2], a_1<a_2;\\
	&t\in[t_0,t_1];\\
	&H=\{x=c\}.
\end{aligned}
\right.
\end{equation}


\subsubsection{Вычисляем первое множество, до переключения}

Вычислим множество достижимости $\XX_1$ первого уравнения (системы) \eqref{eq:prim1a}
Для $\XX_1$ имеем:
$$
\XX[t_1]=\left(\XX_0\dot {-}\intd{t_1}{t_0}\mathcal{Q}(s)ds\right)+\intd{t_0}{t_1}\mathcal{P}(s)ds
$$
Пусть
$$
\tilde{p_1}=\intd{t_0}{t_1}p_1(s)ds;\;\;\; \tilde{p_2}=\intd{t_0}{t_1}p_2(s)ds;
$$

$$
\tilde{q_1}=\intd{t_1}{t_0}q_2(s)ds;\;\;\; \tilde{q_2}=\intd{t_1}{t_0}q_1(s)ds;
$$

$$
\mathcal{Q}(\tau_1,\tau_2)=\intd{\tau_1}{\tau_2}\mathcal{Q}(s)ds
$$
$$
\mathcal{P}(\tau_1,\tau_2)=\intd{\tau_1}{\tau_2}\mathcal{P}(s)ds
$$
Тогда $\mathcal{P}(t_0,t_1)=[\tilde{p_1},\tilde{p_2}]$, $\mathcal{Q}(t_1,t_0)=[\tilde{q_1},\tilde{q_2}]$
Для множества $C=\left(\XX_0\dot{-}\mathcal{Q}(t_1,t_0)\right)$ имеем выражение
$$
\supfn{\ell}{C}=\conv\left[\supfn{\ell}{\XX_0}-\supfn{\ell}{\mathcal{Q}(t_1,t_0)}\right](\ell)
$$

Геометрическую разность можно еще выразить как 
$$
A\dot- B=\bigcap\limits_{\forall b\in B} (A-b)
$$

Множество $C=\XX_0 \dot {-}\mathcal{Q}(t_1,t_0)$, где   $\mathcal{Q}(t_1,t_0)=[\tilde{q}_1,\tilde{q}_2]$, находится как 
$C=[a_1-\tilde{q}_1,a_2-\tilde{q}_2]$, при условии, что $a_2-a_1\ge\tilde{q}_2-\tilde{q}_1$. Тогда 
$$
\XX[t_1]=[a_1-\tilde{q}_1+\tilde{p}_1,a_2-\tilde{q}_2+\tilde{p}_2],\forall t_1:{a_2-a_1\ge\tilde{q}_2-\tilde{q}_1}
$$

Вычислим тоже самое через опорные функции.

\begin{equation} \label{eq:supx1}
\supfnl{\XX[t_1]}=\supfnl{\XX_0\dot {-} \mathcal{Q}(t_1,t_0)}+\supfnl{\mathcal{P}(t_0,t_1)}
\end{equation}

$$
\supfnl{\mathcal{Q}(t_1,t_0)}=\supfnl{\intd{t_1}{t_0}\mathcal{Q}(s)ds}=
\begin{cases}
\ell\tilde{q}_2=\intd{t_1}{t_0}\ell q_1(s)ds,\,\ell\ge 0\\
\ell\tilde{q}_1=\intd{t_1}{t_0}\ell q_2(s)ds,\,\ell<0 \\
\;t_1>t_0,\; q_1(s)\le q_2(s), \forall s
\end{cases}. 
$$
Здесь всегда выполнено $\tilde{q}_1(t_0)\le \tilde{q}_2(t_0)$. 

Для $\rdim{1}$ имеем: $\XX_1=[b_1,b_2],\;\XX_0=[a_1,a_2]$, $\mathcal{P}(s)=[p_1(s),p_2(s)]$, $\mathcal{Q}=[q_1(s),q_2(s)$, $V(\tau)=[\tilde{q}_1(\tau),\tilde{q_2}]$.

Выражение \eqref{eq:supx1} преобразуется к 
$$
\begin{aligned}
\ell=1&: b_2= a_2-\tilde{q}_2(t_0) + \tilde{p}_2(t_1)\\
\ell=-1&: -b_1= -a_1+\tilde{q}_1(t_0) - \tilde{p}_1(t_1)
\end{aligned}
$$

И получаем тоже выражение $\XX_1=[b_1(t_1),b_2(t_2)]=[a_1-\tilde{q}_1(t_0) + \tilde{p}_1(t_1), a_2-\tilde{q}_2(t_0) + \tilde{p}_2(t_1)]$.
Здесь $\conv(\supfnl{\XX_0}\dot {-} \mathcal{Q}(t_1,t_0)) \equiv \supfnl{\XX_0} - \supfnl{\mathcal{Q}(t_1,t_0)}$.

Поскольку $\XX_1$ -- гарантированная оценка, то она обладает тем свойством, что не зависит от помехи. Это множество позволяет нам не рассматривать зависимость от помехи $v_1$. 



\subsubsection{Множество $\{\tau,\XX_H[\tau]\}$}
То, что получаем при пересечении гиперплоскости $H$:
$$
\{(\tau,\XX_H)\}=\{ (\tau,\XX_1[\tau]\cap H)\mid \tau: H\cap\XX_1(\tau,t_0,\XX_0)\neq\emptyset\}
$$

Пусть требуется найти  $(A+B)\cap H$. Представим $A=\bigcup\limits_{i\in I}(a_{\perp}^i+A_{\parallel}^i)$
и $B=\bigcup\limits_{i\in I}(b_{\perp}+B_{\parallel}^i)$, где $a_{\perp}^i\perp H,\;b_{\perp}^i\perp H$ и справедливо $(a\in A^i_\parallel, a\in H) \Rightarrow A^i_\parallel \subset H$, аналогочно для $B^i_\parallel$. Тогда
$$ 
(A+B)\cap H = \bigcup(A_{\parallel}^i+B_{\parallel}^j+a_{\perp}^i+b_{\perp}^j \mid \forall i \in I, j(i)\in I: a_{\perp}^i+b_{\perp}^j\in H) 
$$

Пусть требуется найти множество $B=\{\xi\}$, такое что $\forall \xi\in B: (\xi+A)\cap H \subset\neq\emptyset, \subset X_H$. Выполним разложение $B=\bigcup\limits_{\forall i\in I}(\xi^i=\xi_{\perp}^i+B_\parallel^i$, и соответственно для $A$. Найдем $\xi_\perp^i: (\xi_\perp^i+A_\parallel^i) \cap H \neq \emptyset$. $H={x\mid \ts{x,c}=q}$. Тогда $d^i=\ts{\xi_\perp^i,c}=q-\ts{a_\perp^i,c}$ и $\xi_\perp^i=\frac{d^ic}{\ts{c,c}}$. Теперь найдем $\{B_{\parallel}^j\}$, $\{\xi_\perp^j\}$,$j\in\mathcal{J}\subset I$ для которых выполнено вложение 

$$
\mathcal{J}=\{ i\in I \mid  \exists B_\parallel^i\neq\emptyset: \xi_\perp+B_\parallel^i+A_\parallel^i\subseteq X_H\}
$$

Поскольку $\xi_\perp^i+A_\parallel^i\subset H$ по построению всегда выполнено, то фактически \\ $B_\parallel^i=X_H\dot-(A_\parallel^i+\xi_\perp^i)=(X_H-\xi_\perp^i)\dot-A_\parallel^i$.

Для $\rdim{1}$ есть особенность, заключающаяся в том, что $X_H\cap H=H=x_H\in\rdim{1}, X_H\cap H\neq \emptyset$. 
$$
A\cap H\subset X_H \Leftrightarrow A\cap H  = H \Leftrightarrow H\subset A
$$


Для $\XX_1^{(1)}(\tau)=\{x_\perp\}=[b_1,b_2]$ множество моментов времени пересечения гиперплоскости $\{\tau_H\}=\{\tau\mid H=\{c\}, c\in [b_1(\tau),b_2(\tau)]\}$. Множество $\mathcal{T}=\{\tau_H\}$ - это множество моментов времени не зависящее от помех. В силу условия трансверсальности это множество односвязно.
